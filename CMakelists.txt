cmake_minimum_required(VERSION 3.0)

project(first_laba)

include(FetchContent)
FetchContent_Declare(
  googletest
  GIT_REPOSITORY https://github.com/google/googletest.git
  GIT_TAG v1.15.0
  TLS_VERIFY false
)

set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
FetchContent_MakeAvailable(googletest)

set(SOURCE_EXE main.cpp)

add_library(my_pow_lib my_pow/pow.cpp)
target_include_directories(my_pow_lib PUBLIC my_pow_lib)

add_library(ans_lib ans/ans.cpp)
target_include_directories(ans_lib PUBLIC ans_lib)

target_link_libraries(ans_lib PRIVATE my_pow_lib)

add_executable(run_file ${SOURCE_EXE})
target_link_libraries(run_file PRIVATE my_pow_lib ans_lib)

enable_testing()

add_executable(tests test/tests.cpp)
target_link_libraries(tests ans_lib gtest_main)

add_test(NAME MyProjectTests COMMAND tests)
